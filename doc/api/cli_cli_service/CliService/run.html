run method

Future<void> run()

Starts the CLI service, awaiting user input for commands.

Listens for commands such as 'add', 'list', 'delete', and 'quit' to interact with the notes. This method runs asynchronously and will keep running until the user inputs 'quit'.

Implementation

Future<void> run() async {
  while (true) {
    print('What would you like to do? (add/list/delete/quit)');
    final command = stdin.readLineSync();

    await switch (command) {
      'add' => _addNote(),
      'list' => _listNotes(),
      'delete' => _deleteNote(),
      // ignore: avoid_redundant_argument_values
      'quit' => Future.value(null),
      _ => Future(() {
          print('Unknown command');
        }),
    };

    if (command == 'quit') {
      break;
    }
  }
}
